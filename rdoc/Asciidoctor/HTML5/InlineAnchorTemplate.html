def anchor(target, text, type, document, window = nil)
  case type
  when :xref
    text = document.references[:ids].fetch(target, "[#{target}]") if text.nil?
    %Q(<a href="##{target}">#{text}</a>)
  when :ref
    %Q(<a id="#{target}"></a>)
  when :link
    %Q(<a href="#{target}"#{window && " target=\"#{window}\""}>#{text}</a>)
  when :bibref
    %Q(<a id="#{target}"></a>[#{target}])
  end
end
