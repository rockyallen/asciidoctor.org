def self.puts_indented(level, *args)
  indentation = " " * level * 2

  args.each do |arg|
    self.debug { "#{indentation}#{arg}" }
  end
end

def self.show_debug_output?
  @show_debug || (ENV['DEBUG'] == 'true' && ENV['SUPPRESS_DEBUG'] != 'true')
end
